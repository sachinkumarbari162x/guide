App.tsx (React) 
import React, { useState, useEffect } from 'react';

function App() {
  const [users, setUsers] = useState([]);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    fetch('/api/users')
      .then(res => res.json())
      .then(data => {
        setUsers(data);
        setLoading(false);
      });
  }, []);

  if (loading) return <div>Loading...</div>;

  return (
    <div className="app">
      <h1>Users</h1>
      {users.map(user => (
        <UserCard key={user.id} user={user} />
      ))}
    </div>
  );
}

MainActivity.kt (Compose) 
@Composable
fun UserListScreen(viewModel: UserViewModel = viewModel()) {
    val users by viewModel.users.collectAsState()
    val isLoading by viewModel.isLoading.collectAsState()
    
    Scaffold(
        topBar = { TopAppBar(title = { Text("Users") }) }
    ) { padding ->
        if (isLoading) {
            CircularProgressIndicator(
                modifier = Modifier.fillMaxSize().wrapContentSize()
            )
        } else {
            LazyColumn(contentPadding = padding) {
                items(users) { user ->
                    UserCard(user = user)
                }
            }
        }
    }
}

import SwiftUI

struct UserListView: View {
    @StateObject private var viewModel = UserViewModel()
    
    var body: some View {
        NavigationStack {
            Group {
                if viewModel.isLoading {
                    ProgressView("Loading...")
                } else {
                    List(viewModel.users) { user in
                        UserRow(user: user)
                    }
                }
            }
            .navigationTitle("Users")
            .task {
                await viewModel.fetchUsers()
            }
        }
    }
}

Section 05

App Architecture

Patterns and best practices for building scalable applications.

MVC / MVP / MVVM

Separation of concerns patterns.

  • MVC: Model-View-Controller
  • MVP: Model-View-Presenter
  • MVVM: Model-View-ViewModel
  • Choose based on platform

Clean Architecture

Dependency rule & layers.

  • Domain layer (use cases)
  • Data layer (repositories)
  • Presentation layer (UI)
  • Testable and maintainable

Unidirectional Data Flow

Predictable state updates.

  • Actions trigger changes
  • State flows down
  • Events flow up
  • Redux, MVI patterns

Local State

  • React: useState, useReducer
  • SwiftUI: @State, @Binding
  • Compose: remember { mutableStateOf() }

Global State

  • React: Redux, Zustand, Context
  • Flutter: Riverpod, Provider
  • Android: ViewModel + StateFlow

Server State

  • React Query / TanStack Query
  • SWR for data fetching
  • Apollo Client for GraphQL

Unit Testing

  • Jest (JavaScript)
  • XCTest (Swift)
  • JUnit (Kotlin)
  • Test business logic

UI Testing

  • React Testing Library
  • XCUITest (iOS)
  • Espresso (Android)
  • Cypress (E2E)

Snapshot Testing

  • Capture UI state
  • Detect visual regressions
  • Storybook for components
